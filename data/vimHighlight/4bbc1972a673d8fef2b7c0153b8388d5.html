1 # Import `os` & `shutil` module  2 import os, shutil  3   4 # Get Current Working Directory  5 search_dir = os.getcwd()  6   7 # Goto that Directory  8 os.chdir(search_dir)  9  10 # List Files in the Directory, don't list Subdirectories 11 files = filter(os.path.isfile, os.listdir(search_dir)) 12  13 # For each file, store it in an `files` list 14 files = [os.path.join(search_dir, f) for f in files] 15  16 # Sort all files according to the `date`  17 files.sort(key=lambda x: os.path.getmtime(x)) 18  19 # Directory name to put your sorted files in 20 dir = "sorted" 21  22 # Create that directory 23 os.system("sudo mkdir -p " + dir) 24  25 # Extension of the file, mine are images 26 extension=".jpg" 27  28 # Starting name 29 i=1 30  31 # For every file copy it in sorted directory 32 for filename in files: 33         destination = "sorted/" + str(i) + extension 34         shutil.copy(filename,destination) 35         i = i + 1 36 print "Copied all files to 'sorted' directory" 37 # End of Program
